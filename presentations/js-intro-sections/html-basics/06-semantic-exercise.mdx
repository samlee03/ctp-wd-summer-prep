// Test the semantic HTML structure
describe('Semantic HTML Tests', () => {
    it('should use header element for site header', () => {
        const header = document.querySelector('header');
        assert.exists(header, 'Page should have a <header> element');
        assert.exists(header.querySelector('h1'), 'Header should contain an h1');
    });
    
    it('should use nav element for navigation', () => {
        const nav = document.querySelector('nav');
        assert.exists(nav, 'Page should have a <nav> element');
        assert.exists(nav.querySelector('ul'), 'Nav should contain a list');
        assert.isAtLeast(nav.querySelectorAll('a').length, 3, 'Nav should have proper links');
    });
    
    it('should use main element for primary content', () => {
        const main = document.querySelector('main');
        assert.exists(main, 'Page should have a <main> element');
    });
    
    it('should use article element for blog post', () => {
        const article = document.querySelector('article');
        assert.exists(article, 'Blog post should use <article> element');
        assert.exists(article.querySelector('h2'), 'Article should have a heading');
    });
    
    it('should use time element with datetime', () => {
        const time = document.querySelector('time');
        assert.exists(time, 'Should use <time> element for dates');
        assert.exists(time.getAttribute('datetime'), 'Time element should have datetime attribute');
    });
    
    it('should use aside for sidebar content', () => {
        const aside = document.querySelector('aside');
        assert.exists(aside, 'Sidebar should use <aside> element');
    });
    
    it('should use semantic buttons instead of spans', () => {
        const buttons = document.querySelectorAll('button');
        assert.isAtLeast(buttons.length, 2, 'Interactive elements should be buttons');
        
        const clickableSpans = document.querySelectorAll('span[onclick]');
        assert.equal(clickableSpans.length, 0, 'Should not use spans with onclick');
    });
    
    it('should use footer element', () => {
        const footer = document.querySelector('footer');
        assert.exists(footer, 'Page should have a <footer> element');
    });
    
    it('should have proper document structure', () => {
        assert.exists(document.querySelector('html[lang]'), 'HTML should have lang attribute');
        assert.exists(document.querySelector('meta[charset]'), 'Should have charset meta tag');
    });
});

<Notes>
This exercise challenges students to transform poorly structured HTML into semantic, accessible markup. The starting code represents common mistakes developers make when focusing only on visual appearance rather than document structure.

The exercise reveals several anti-patterns. Generic divs are used for everything, making the document structure meaningless to assistive technologies and search engines. Clickable spans with onclick handlers exclude keyboard users and aren't announced as interactive by screen readers. The lack of heading hierarchy makes navigation difficult for screen reader users.

Students must identify each element's purpose and choose appropriate semantic replacements. The "top-section" should become a header element. The "site-title" needs to be an h1 heading. The navigation menu requires nav element with proper links, not clickable spans. Blog posts should use article elements with proper heading hierarchy.

This hands-on practice reinforces that HTML isn't just about visual structure - it's about creating meaningful documents. Students learn to think about their users: Can a screen reader user navigate this? Can a keyboard user interact with everything? Does the document make sense without CSS? These questions lead to better, more inclusive web development practices.
</Notes>
